css针对IE浏览器设定

By Cheney Fan

4月 29 2014 更新日期:6月 27 2014

文章目錄

以前没怎么关心过这个问题。看到一个比较好的解决方案是：先在javascript中判断IE版本，然后将版本信息加入html元素的class中。

function getIEVersion() {
      var rv = -1;
      if (navigator.appName == 'Microsoft Internet Explorer') {
          var ua = navigator.userAgent;
          var re = new RegExp("MSIE ([0-9]{1,}[\.0-9]{0,})");
          if (re.exec(ua) != null)
          rv = parseFloat(RegExp.$1);
      }
      else if(navigator.appName == 'Netscape'){
          var ua = navigator.userAgent;
          var re = new RegExp("Trident/.*rv:([0-9]{1,}[\.0-9]{0,})"); //for IE 11
          if (re.exec(ua) != null)
          rv = parseFloat(RegExp.$1);
      }
      return rv;
  }
  IEV = getIEVersion();
  if(IEV>=6&&IEV<9){
      window.location.href="http://no-surport.***.com"; //IE6~8就不支持了  
  }else if(IEV>=9){
      document.documentElement.className+=' ie ie'+IEV; //将ie及版本信息加入html的class中。
  }

接着在css里面进行设置：

.ie9 .fiterMask{
    filter: progid:DXImageTransform.Microsoft.Blur(PixelRadius=10, MakeShadow=false);
}
.ie .loginDiv{
    left:50em;
}
